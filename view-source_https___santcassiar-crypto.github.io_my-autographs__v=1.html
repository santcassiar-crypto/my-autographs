<script>
const KEY="vaultos_v4";
let DB=[];

function save(){localStorage.setItem(KEY,JSON.stringify(DB));render();}
function load(){DB=JSON.parse(localStorage.getItem(KEY)||"[]");render();}

function importJSON(){
try{
DB=JSON.parse(document.getElementById('jsonBox').value);
save();
alert("Imported "+DB.length+" records");
}catch(e){alert("Invalid JSON");}
}

function exportJSON(){
const b=new Blob([JSON.stringify(DB,null,2)],{type:'application/json'});
const a=document.createElement('a');
a.href=URL.createObjectURL(b);
a.download='vault_backup.json';
a.click();
}

function stats(){
return {
total:DB.length,
success:DB.filter(x=>x.Status==="SUCCESS").length,
pending:DB.filter(x=>x.Status==="PENDING").length,
rts:DB.filter(x=>(x.Status||"").includes("RTS")).length
}
}

function score(r){
let s=50;
if((r.Status||"")==="PENDING") s+=20;
if((r.Name||"").match(/England|Arsenal|training|cast/i)) s-=10;
return s;
}

function render(){
let st=stats();
document.getElementById('stats').innerHTML=`
<div class="card"><h2>${st.total}</h2>Total</div>
<div class="card"><h2>${st.success}</h2>Success</div>
<div class="card"><h2>${st.pending}</h2>Pending</div>
<div class="card"><h2>${st.rts}</h2>RTS</div>`;

let top=DB.filter(x=>x.Status==="PENDING")
.sort((a,b)=>score(b)-score(a))
.slice(0,10);

document.getElementById('intel').innerHTML =
"<b>Top 10 Likely Returns</b><ol>"+
top.map(x=>`<li>${x.Name} (${score(x)}%)</li>`).join("")+
"</ol>";

let q=(document.getElementById('search').value||'').toLowerCase();

document.getElementById('list').innerHTML =
DB.filter(x=>(x.Name||'').toLowerCase().includes(q))
.map(x=>`
<div class="card">
<b>${x.Name||''}</b><br>
<span class="small">${x.Status||''}</span>
</div>`).join("");
}

document.getElementById('search').addEventListener('input',render);
load();
</script>
